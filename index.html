Formeln:

$$ \begin{align} B(z) &= \frac{\mu_0}{4\cdot\pi} \cdot \frac{2\cdot\pi\cdot N \cdot I \cdot R^2 \cdot \hat{z}}{(R^2+z^2)^\frac{3}{2}} \\ \end{align} $$

$$ \begin{align} H(z) &= \frac{B(z)}{\mu_0}\\ \end{align} $$

$$ \begin{align} U(z)=-m(z)\cdot B(z) \end{align} $$

$$ \begin{align} F(z)&=-\nabla U(z)=\nabla(m(z)\cdot B(z)) \\ &= \frac{d}{dz}(m(z)\cdot B(z)) \end{align} $$

$$ \begin{align} \chi &= \mu_r - 1 \end{align} $$

$$ \begin{align} M(z) &= \chi\cdot H(z) \\ \end{align} $$

$$ \begin{align} V = \frac{4}{3} \cdot \pi \cdot r^3 \end{align} $$

$$ \begin{align} m &= V\cdot M \end{align} $$
